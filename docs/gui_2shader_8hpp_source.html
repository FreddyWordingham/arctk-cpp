332  {

334 
335  for (size_t i = 0; i < uniform_names_.size(); ++i)
336  {
337  PRE(!uniform_names_[i].empty());
338  PRE(uniform_names_[i] != "model");
339  PRE(uniform_names_[i] != "view");
340  PRE(uniform_names_[i] != "proj");
341 
342  if (uniforms.find(uniform_names_[i]) != uniforms.end())

344  std::cerr << "Unable to construct gui shader.\n"

346 
347  std::exit(exit::error::SHADER_UNIFORM_DUPLICATE);
348  }

351  }
