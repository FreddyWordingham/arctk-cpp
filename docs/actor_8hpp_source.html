149  inline Actor::Actor(const std::vector<GLfloat>& verts_, const std::vector<size_t>& layout_, const GLenum primitive_type_, const GLenum fill_mode_) noexcept
150  : _num_vert(static_cast<GLsizei>(verts_.size() / math::container::sum(layout_)))
151  , _vao(init_vao())
152  , _vbo(init_vbo())
153  , _primitive_type(primitive_type_)
154  , _fill_mode(fill_mode_)
155  , _trans(glm::vec3(0.0f, 0.0f, 0.0f))
156  , _rotate(glm::vec3(0.0f, 0.0f, 0.0f))
157  , _scale(glm::vec3(1.0f, 1.0f, 1.0f))
158  {
159  PRE((verts_.size() % math::container::sum(layout_)) == 0);
160 
161  glBindVertexArray(_vao);
162  glBindBuffer(GL_ARRAY_BUFFER, _vbo);
163  glBufferData(GL_ARRAY_BUFFER, static_cast<GLsizeiptr>(verts_.size() * sizeof(GLfloat)), &verts_.front(), GL_STATIC_DRAW);
164 
165  const size_t chunk_size = math::container::sum(layout_) * sizeof(GLfloat);
166  size_t start = 0;
167  for (size_t i = 0; i < layout_.size(); ++i)
168  {
169  glVertexAttribPointer(static_cast<GLuint>(i), static_cast<GLint>(layout_[i]), GL_FLOAT, GL_FALSE, static_cast<GLsizei>(chunk_size),
170  reinterpret_cast<GLvoid*>(start * sizeof(GLfloat))); // NOLINT
171  glEnableVertexAttribArray(static_cast<GLuint>(i));
172 
173  start += layout_[i];
174  }
175 
176  glBindBuffer(GL_ARRAY_BUFFER, 0);
177 
178  glBindVertexArray(0);
179 
180  update_model();
181  }
