470  {
471  PRE(min_.x < max_.x);

473  PRE(cell_size_.x > 0.0f);
474  PRE(cell_size_.y > 0.0f);
475 
476  const auto x_start = static_cast<int>(std::trunc(min_.x / cell_size_.x));
477  const auto x_end = static_cast<int>(std::trunc(max_.x / cell_size_.x));
478  const auto y_start = static_cast<int>(std::trunc(min_.y / cell_size_.y));
479  const auto y_end = static_cast<int>(std::trunc(max_.y / cell_size_.y));
480 
481  std::vector<glm::vec3> verts;
482  verts.reserve(2 * static_cast<size_t>((x_end - x_start + 1) + (y_end - y_start + 1)));
483 
484  for (int i = x_start; i <= x_end; ++i)
485  {
486  verts.emplace_back(glm::vec3(static_cast<float>(i) * cell_size_.x, min_.y, 0.0f));
487  verts.emplace_back(glm::vec3(static_cast<float>(i) * cell_size_.x, max_.y, 0.0f));
488  }
489 
490  for (int i = y_start; i <= y_end; ++i)
491  {
492  verts.emplace_back(glm::vec3(min_.x, static_cast<float>(i) * cell_size_.y, 0.0f));
493  verts.emplace_back(glm::vec3(max_.x, static_cast<float>(i) * cell_size_.y, 0.0f));
494  }
495 
496  Actor grid(verts, {3}, GL_LINES);
497  grid.set_col(glm::vec3(0.25f, 0.25f, 0.25f));
498 
499  return (grid);
