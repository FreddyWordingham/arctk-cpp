468  {
469  PRE(min_.x < max_.x);

471  PRE(cell_size_.x > 0.0f);
472  PRE(cell_size_.y > 0.0f);
473 
474  const auto x_start = static_cast<int>(std::trunc(min_.x / cell_size_.x));
475  const auto x_end = static_cast<int>(std::trunc(max_.x / cell_size_.x));
476  const auto y_start = static_cast<int>(std::trunc(min_.y / cell_size_.y));
477  const auto y_end = static_cast<int>(std::trunc(max_.y / cell_size_.y));
478 
479  std::vector<glm::vec3> verts;
480  verts.reserve(2 * static_cast<size_t>((x_end - x_start + 1) + (y_end - y_start + 1)));
481 
482  for (int i = x_start; i <= x_end; ++i)
483  {
484  verts.emplace_back(glm::vec3(static_cast<float>(i) * cell_size_.x, min_.y, 0.0f));
485  verts.emplace_back(glm::vec3(static_cast<float>(i) * cell_size_.x, max_.y, 0.0f));
486  }
487 
488  for (int i = y_start; i <= y_end; ++i)
489  {
490  verts.emplace_back(glm::vec3(min_.x, static_cast<float>(i) * cell_size_.y, 0.0f));
491  verts.emplace_back(glm::vec3(max_.x, static_cast<float>(i) * cell_size_.y, 0.0f));
492  }
493 
494  Actor grid(verts, {3}, GL_LINES);
495  grid.set_col(glm::vec3(0.25f, 0.25f, 0.25f));

497  return (grid);
