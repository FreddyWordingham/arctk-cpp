147  inline Actor::Actor(const std::vector<GLfloat>& verts_, const std::vector<size_t>& layout_, const GLenum primitive_type_, const GLenum fill_mode_) noexcept
148  : _num_vert(static_cast<GLsizei>(verts_.size() / math::sum(layout_)))
149  , _vao(init_vao())
150  , _vbo(init_vbo())
151  , _primitive_type(primitive_type_)
152  , _fill_mode(fill_mode_)
153  , _trans(glm::vec3(0.0f, 0.0f, 0.0f))
154  , _rotate(glm::vec3(0.0f, 0.0f, 0.0f))
155  , _scale(glm::vec3(1.0f, 1.0f, 1.0f))
156  {
157  assert((verts_.size() % math::sum(layout_)) == 0);
158 
159  glBindVertexArray(_vao);
160  glBindBuffer(GL_ARRAY_BUFFER, _vbo);
161  glBufferData(GL_ARRAY_BUFFER, verts_.size() * sizeof(GLfloat), &verts_.front(), GL_STATIC_DRAW);
162 
163  const size_t chunk_size = math::sum(layout_) * sizeof(GLfloat);
164  size_t start = 0;
165  for (size_t i = 0; i < layout_.size(); ++i)
166  {
167  glVertexAttribPointer(i, layout_[i], GL_FLOAT, GL_FALSE, chunk_size,
168  reinterpret_cast<GLvoid*>(start * sizeof(GLfloat))); // NOLINT
169  glEnableVertexAttribArray(i);
170 
171  start += layout_[i];
172  }
173 
174  glBindBuffer(GL_ARRAY_BUFFER, 0);
175 
176  glBindVertexArray(0);
177 
178  update_model();
179  }
