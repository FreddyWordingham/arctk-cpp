183  {
184  glfwPollEvents();
185 
186  if ((glfwGetKey(win_.handle(), QUIT_KEY) == GL_TRUE) || (glfwWindowShouldClose(win_.handle()) == GL_TRUE))
187  {
188  return (false);
189  }
190 
191  for (auto& [key, keybind] : _map)
192  {
193  const int state = glfwGetKey(win_.handle(), key);
194 
195  if (state == GLFW_PRESS)
196  {
197  if (keybind.sticky() || (keybind.state() == GLFW_RELEASE))
198  {
199  keybind();
200  }
201  }
202 
203  keybind.set_state(state);
204  }
205 
206  return (true);
