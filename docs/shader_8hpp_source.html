326  {
327  std::map<std::string, GLint> uniforms;
328 
329  for (size_t i = 0; i < uniform_names_.size(); ++i)
330  {
331  PRE(!uniform_names_[i].empty());
332  PRE(uniform_names_[i] != "model");
333  PRE(uniform_names_[i] != "view");
334  PRE(uniform_names_[i] != "proj");
335 
336  if (uniforms.find(uniform_names_[i]) != uniforms.end())
337  {
338  std::cerr << "Unable to construct gui shader.\n"

340 
341  std::exit(exit::error::SHADER_UNIFORM_DUPLICATE);
342  }
343

345  }
