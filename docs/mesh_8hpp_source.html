171  }
172  else if (word == NORM_KEYWORD)

174  vec4 norm;
175  line_stream >> norm.x >> norm.y >> norm.z;
176  norm.w = 0.0;

179

181  }
182

186  << "Unable to parse line: `" << line << "`.\n";
187  }
188  }

191  while (std::getline(serial_stream, line))
192  {

194  std::string word;
195  line_stream >> word;
196 
197  if (word == FACE_KEYWORD)
198  {
199  ++num_faces;

201  std::array<std::string, 3> face;
202  line_stream >> face[index::vertex::ALPHA] >> face[index::vertex::BETA] >> face[index::vertex::GAMMA];
203 
204  if (line_stream.rdbuf()->in_avail() != 0)

208 
209  std::exit(exit::error::FAILED_PARSE);

216  const size_t last_slash = face[i].find_last_of('/');
