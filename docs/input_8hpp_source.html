112

121  i = string_container_end_pos;

206  inline void extract_contents(std::string* const str_, const consts::format::container type_) noexcept
207  {
208  PRE(str_ != nullptr);
209 
210  std::string& str_ref = *str_;
211 
212  const size_t open = str_ref.find_first_of(consts::format::OPENERS[type_]);
213  if (open == std::string::npos)
214  {
215  std::cerr << "Error extracting contents of container.\n"
216  << "Containter opener was not found.\n";
217 
218  std::exit(exit::error::FAILED_PARSE);
219  }
220  else if (open != 0)
221  {
222  std::cerr << "Error extracting contents of container.\n"
223  << "Additional characters found before container opener.\n";
224 
225  std::exit(exit::error::FAILED_PARSE);
226  }
227  str_ref.erase(open, 1);
228 
229  const size_t close = str_ref.find_last_of(consts::format::CLOSERS[type_]);
230  if (close == std::string::npos)
231  {
232  std::cerr << "Error extracting contents of container.\n"
233  << "Containter closer was not found.\n";
234 
235  std::exit(exit::error::FAILED_PARSE);
236  }
237  else if (close != (str_ref.size() - 1))
238  {
239  std::cerr << "Error extracting contents of container.\n"
240  << "Additional characters found after container closer.\n";
241 
242  std::exit(exit::error::FAILED_PARSE);
243  }
244  str_ref.erase(close, 1);
245  }

254  inline std::vector<std::string> tokenise(const std::string& str_) noexcept
255  {
256  std::vector<std::string> tokens;
257 
258  size_t prev_delim = 0;
259  for (size_t i = 0; i < str_.size(); ++i)
260  {
261  if (str_[i] == consts::format::DELIM)
262  {
263  tokens.emplace_back(str_.substr(prev_delim, i - prev_delim));
264  prev_delim = i + 1;
265  }
266  else
267  {
268  for (size_t j = 0; j < consts::format::OPENERS.size(); ++j)
269  {
270  if (str_[i] == consts::format::OPENERS[j])
271  {
272  i = str_.find_first_of(consts::format::CLOSERS[j], i + 1);
273 
274  if (i == std::string::npos)
275  {
276  std::cerr << "Error parsing container.\n"
277  << "Corresponding container closer was not found.\n";
278 
279  std::exit(exit::error::FAILED_PARSE);
280  }
281 
282  break;
283  }
284  }
285  }
286  }
287  tokens.emplace_back(str_.substr(prev_delim, str_.size()));
288 
289  return (tokens);
290  }
